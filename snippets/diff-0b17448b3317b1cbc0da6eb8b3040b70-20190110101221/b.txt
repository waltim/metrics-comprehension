      private static List makeSVMLightLineInfos(List lines) {        List lineInfos = new ArrayList<>(lines.size());          lines              .stream()              .map(                  line -> {                    line = line.replaceFirst("#.*$", "");                    return line;                  })              .forEach(                  line -> {                    lineInfos.add(line.split("\\s+"));                  });        return lineInfos;      }
