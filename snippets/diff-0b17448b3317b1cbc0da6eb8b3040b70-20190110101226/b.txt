      private void addAllInterningAndPrefixing(Collection accumulator, Collection addend, String prefix) {        assert prefix != null;          addend              .stream()              .map(                  protoFeat -> {                    if (!prefix.isEmpty()) {                      protoFeat = prefix + protoFeat;                    }                    return protoFeat;                  })              .map(                  protoFeat -> {                    if (globalFlags.intern) {                      protoFeat = protoFeat.intern();                    }                    return protoFeat;                  })              .forEach(                  protoFeat -> {                    accumulator.add(protoFeat);                  });        }
